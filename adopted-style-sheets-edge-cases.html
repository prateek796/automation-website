adoptedStyleSheets override — edge cases

Tests for VWO's installAdoptedStyleSheetsOverride on Document.prototype and ShadowRoot.prototype. The override is installed only when a prototype is resolved via global constructor or native fallback. 

// Document
const NativeDocument = NativeConstants.get('Document');
const documentPrototype =
  (typeof Document !== 'undefined' && Document.prototype) ||
  (NativeDocument && NativeDocument.prototype);
if (documentPrototype) {
  installAdoptedStyleSheetsOverride(documentPrototype, context, NativeArray, false);
}

// ShadowRoot (isShadow = true)
const NativeShadowRoot = NativeConstants.get('ShadowRoot');
const shadowRootPrototype =
  (typeof ShadowRoot !== 'undefined' && ShadowRoot.prototype) ||
  (NativeShadowRoot && NativeShadowRoot.prototype);
if (shadowRootPrototype) {
  installAdoptedStyleSheetsOverride(shadowRootPrototype, context, NativeArray, true);
}
